# 三体模拟经营游戏框架设计

## 设计目标
- **轻量级**：适合个人开发，不过度工程化
- **可扩展**：模块清晰，后期可逐步添加功能
- **2D + 轻量3D**：星球运动3D可视化，其他交互2D

## 技术栈

| 类别 | 选择 | 理由 |
|------|------|------|
| 图形渲染 | Pygame | 同时支持2D和简单3D渲染 |
| 数学计算 | NumPy | 3D投影矩阵、三体运动计算 |
| 配置管理 | PyYAML | 人类可读配置文件 |
| 数据存储 | JSON | 存档序列化 |

> **注意**：不需要 ModernGL/OpenGL，手写3D→2D投影足够满足需求

## 3D功能实现方案

### 1. 星球外观渲染

> **开发策略**：先使用方案B，后期可升级到方案C

**方案A：2D圆形（最简）**
```python
def draw_planet(screen, camera, x, y, z, radius, color):
    """根据距离调整大小的2D圆形"""
    sx, sy = camera.world_to_screen(x, y, z)
    scale = camera.fov / (camera.fov + z)
    screen_radius = max(1, int(radius * scale))
    pygame.draw.circle(screen, color, (sx, sy), screen_radius)
```

**方案B：球体感渐变（推荐）**
```python
def draw_sphere(screen, x, y, z, radius, color):
    """多层同心圆模拟球体感"""
    sx, sy = camera.world_to_screen(x, y, z)
    scale = camera.fov / (camera.fov + z)
    screen_radius = max(1, int(radius * scale))

    for r in range(screen_radius, 0, -2):
        # 边缘暗，中心亮，产生球体感
        factor = 1 - (r / screen_radius) ** 2
        bright = tuple(min(255, int(c * (0.3 + 0.7 * factor))) for c in color)
        pygame.draw.circle(screen, bright, (sx, sy), r)
```

**方案C：预渲染纹理（最高质量）**
```python
def draw_textured_planet(screen, camera, planet):
    """加载星球纹理图片，缩放后绘制"""
    sx, sy = camera.world_to_screen(planet.x, planet.y, planet.z)
    scale = camera.fov / (camera.fov + planet.z)
    size = int(planet.radius * scale * 2)

    if size > 0:
        scaled = pygame.transform.scale(planet.texture, (size, size))
        screen.blit(scaled, (sx - size // 2, sy - size // 2))
```

class StarField:
    def __init__(self, count=500, width=1280, height=720):
        # x, y: 屏幕位置, z: 深度(影响闪烁), brightness: 亮度
        self.stars = [
            (random.randint(0, width), random.randint(0, height),
             random.uniform(0.5, 2.0), random.randint(150, 255))

        ]

    def render(self, screen):
        for x, y, z, base_bright in self.stars:
            # 轻微闪烁效果

            color = (bright, bright, bright)
            size = max(1, int(z))

        self.depth = depth  # 深度越大移动越慢

### 3. 自由视角控制

```python
class Camera:
    def __init__(self, position=(0, 0, -500), fov=500):
        self.pos = list(position)      # [x, y, z] 摄像机位置
        self.yaw = 0                   # 水平旋转角度
        self.pitch = 0                 # 垂直旋转角度
        self.fov = fov
        self.speed = 5                 # 移动速度

    def rotate(self, dx, dy):
        """鼠标拖拽旋转视角"""
        self.yaw += dx * 0.005
        self.pitch += dy * 0.005
        # 限制垂直角度，防止翻转
        self.pitch = max(-1.5, min(1.5, self.pitch))

    def move(self, forward=0, right=0, up=0):
        """WASD/方向键移动"""
        # 计算前进方向
        cos_y, sin_y = math.cos(self.yaw), math.sin(self.yaw)

        self.pos[0] += right * cos_y - forward * sin_y
        self.pos[2] += right * sin_y + forward * cos_y
        self.pos[1] += up

    def zoom(self, delta):
        """滚轮缩放/前后移动"""
        self.pos[2] += delta * 10
```

**输入处理**
```python
def handle_input(events, camera, keys):
    for event in events:
        if event.type == pygame.MOUSEMOTION and pygame.mouse.get_pressed()[0]:
            camera.rotate(event.rel[0], event.rel[1])

        if keys[pygame.K_w]: camera.move(forward=camera.speed)
        if keys[pygame.K_s]: camera.move(forward=-camera.speed)
        if keys[pygame.K_a]: camera.move(right=-camera.speed)
        if keys[pygame.K_d]: camera.move(right=camera.speed)

        if event.type == pygame.MOUSEWHEEL:
            camera.zoom(event.y)
```

## 3D投影实现要点

### 旋转变换矩阵

```python
import numpy as np

def rotate_y(point, angle):
    """绕Y轴旋转（水平视角）"""
    cos_a, sin_a = np.cos(angle), np.sin(angle)
    return np.array([
        point[0] * cos_a + point[2] * sin_a,
        point[1],
        -point[0] * sin_a + point[2] * cos_a
    ])

def rotate_x(point, angle):
    """绕X轴旋转（垂直视角）"""
    cos_a, sin_a = np.cos(angle), np.sin(angle)
    return np.array([
        point[0],
        point[1] * cos_a - point[2] * sin_a,
        point[1] * sin_a + point[2] * cos_a
    ])
```

### 透视投影

```python
def project(x, y, z, fov, screen_center):
    """透视投影到2D"""
    if z + fov <= 0:
        return None  # 点在摄像机后方
    scale = fov / (fov + z)
    return (
        screen_center[0] + x * scale,
        screen_center[1] + y * scale
    )
```
